\[ g(x) = ax \]
\[ a < 0 \]

\[ f(x_0) = g(x_0) \quad x_0 \neq 0 \]
\[ a{x_0}^2 - 2ax_0 = ax_0 \]
\[ a{x_0}^2 - 3ax_0 = 0 \]
\[ x_{0_{1,2}} = \frac{3a \pm \sqrt{9a^2}}{2a} = \frac{3a \pm 3a}{2a} = \frac{3 \pm 3}{2} \]

\[ = \frac{a{3}^3}{3} - \frac{3a{3}^2}{2} = 9a - \frac{27}{2}a = -4.5a \]
\[ \Rightarrow \underline{\underline{a = -\frac{F}{4.5}}} \]
